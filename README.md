```bash
npm install --save @metapriori/react-panzoom
```

```tsx
import React, { Component } from 'react'

import MyComponent from '@metapriori/react-panzoom'
import '@metapriori/react-panzoom/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
